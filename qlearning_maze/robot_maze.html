import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    random_number = random.randint(0, 9)
    if random_number < 10 * epsilon: # 
        # 
        random_number = random.randint(0, 3)
        action = actions[random_number]
    else: 
        #  Q 
        action = actions[0]
        q_value = qline[action]
        for key, value in qline.items():
            if value > q_value:
                q_value = value
                action = key
    return action
